from urllib.request import urlopen
import json

def get_geo_code(area) :
    url = 'https://maps.googleapis.com/maps/api/geocode/json?address=' + area + '&key=AIzaSyB3sfpVnpSzExtr-6g6uBlL1KGVnWrpkF4'
    url = url.replace(' ','+')
    print(url)
    jsonurl = urlopen(url)
    text = json.loads(jsonurl.read())
    latitude = text['results'][0]['geometry']['location']['lat']
    longitude = text['results'][0]['geometry']['location']['lng']
    return latitude, longitude
